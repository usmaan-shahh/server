@id --> PrimaryKey

Primary keys are used in: Joins, Indexes, Foreign keys.
CONSTRAINT? means Optional Null allowed.
npm i -D prisma
npm i @prisma/client
npx prisma generate --> node_modules/prisma/client/

generator client {
provider = "prisma-client-js"
}

datasource db {
provider = "postgresql"
}

model user {
id Int @id @default(autoincrement())
name String
email String @unique
password String
posts Post[]
createdAt DateTime @default(now())
}

model Post {
id Int @id @default(autoincrement())
title String
content String?
authorId Int
author user @relation(fields: [authorId], references: [id])
createdAt DateTime @default(now())
}
